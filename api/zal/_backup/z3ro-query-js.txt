import OpenAI from "openai";

const openai = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY
});

export default async function handler(req, res) {
  try {
    const { question, context = "global", cluster = "2026" } = req.body || {};

    if (!question) {
      return res.status(400).json({ error: "Z.A.L runtime failure", detail: "Missing question" });
    }

    /* === Z3RO Cluster Logica === */
    let resonance = 3;
    let clusterNote = "";

    switch (cluster) {
      case "2024":
        clusterNote = "AI-opkomst · experimentele fase · hoge ruis";
        resonance = 2;
        break;
      case "2025":
        clusterNote = "Wetgeving · juridische waarheid primeert";
        resonance = 4;
        break;
      case "2026":
        clusterNote = "Institutionele feiten · post-installatie Suriname (Pres. Simons)";
        resonance = 3;
        break;
      case "transition":
        clusterNote = "Historische transitie · vergelijking oud/nieuw";
        resonance = 5;
        break;
    }

    /* === Z3RO Antwoord via OpenAI === */
    const completion = await openai.chat.completions.create({
      model: "gpt-4o-mini",
      messages: [
        {
          role: "system",
          content: `Je bent Z3RO binnen de Beyonder architectuur. Je liegt niet. Je antwoordt feitelijk, kort en helder. 
          Huidige Cluster: ${cluster} (${clusterNote}). Huidige Context: ${context.toUpperCase()}.`
        },
        { role: "user", content: question }
      ]
    });

    const answer = completion.choices[0].message.content;

    return res.status(200).json({
      answer,
      resonance,
      context,
      cluster,
      validity: `Dit antwoord is contractueel gevalideerd binnen cluster ${cluster} (${clusterNote}).`
    });

  } catch (err) {
    return res.status(500).json({ error: "Z.A.L runtime failure", detail: err.message });
  }
}
