  <script>
    let currentPersona = 'aetron';

    function setPersona(persona) {
        currentPersona = persona;
        document.querySelectorAll('.persona-btn').forEach(btn => btn.classList.remove('active'));
        document.getElementById('btn-' + persona).classList.add('active');
        const bar = document.getElementById('search-bar');
        if (persona === 'aetron') bar.style.borderColor = '#8b5cf6';
        if (persona === 'z3ro') bar.style.borderColor = '#06b6d4';
        if (persona === 'luxen') bar.style.borderColor = '#f472b6';
    }

    async function sendQuery() {
        const input = document.getElementById('morphic-query');
        const box = document.getElementById('morphic-response');
        const content = document.getElementById('response-content');
        const agentLabel = document.getElementById('response-agent');
        const roleLabel = document.getElementById('response-role');
        
        if(!input.value) return;
        
        box.classList.remove('hidden');
        content.innerHTML = '<span class="text-slate-500 animate-pulse">> Initialiseren van Morphic Handshake...</span>';
        
        if (currentPersona === 'aetron') { box.style.borderColor = '#8b5cf6'; agentLabel.style.color = '#8b5cf6'; }
        if (currentPersona === 'z3ro') { box.style.borderColor = '#06b6d4'; agentLabel.style.color = '#06b6d4'; }
        if (currentPersona === 'luxen') { box.style.borderColor = '#f472b6'; agentLabel.style.color = '#f472b6'; }

        // Echo Animatie
        setTimeout(() => { if(content.innerHTML.includes('Initialiseren')) content.innerHTML = '<span class="text-cyan-600 font-bold">...Resonance [01] ‚Üª Echo 1.9 detected...</span>'; }, 800);

        try {
            const response = await fetch('/api/aetron/scan', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ 
                    query: input.value,
                    manualPersona: currentPersona 
                })
            });

            const data = await response.json();

            if (data.resonance) {
                agentLabel.innerText = data.persona.toUpperCase();
                roleLabel.innerText = data.role;

                // BEELDEN GENEREREN (Als ze er zijn)
                let imagesHtml = '';
                if (data.images && data.images.length > 0) {
                    imagesHtml = `
                        <div class="mt-4 grid grid-cols-2 md:grid-cols-3 gap-2 border-t border-slate-800 pt-4">
                            ${data.images.slice(0, 3).map(url => `
                                <a href="${url}" target="_blank" class="block overflow-hidden rounded border border-cyan-500/30 hover:border-cyan-400 transition group">
                                    <img src="${url}" class="w-full h-24 object-cover opacity-80 group-hover:opacity-100 transition duration-500 hover:scale-110" alt="Visual Data">
                                </a>
                            `).join('')}
                        </div>
                        <p class="text-[9px] text-slate-500 mt-1 uppercase tracking-widest">Visual Data Stream</p>
                    `;
                }

                content.innerHTML = `<span class="typing-effect leading-relaxed block">${data.output}</span>${imagesHtml}<div class="mt-4 text-[9px] text-slate-600 font-mono">‚Üª Timestamp: ${data.timestamp}</div>`;
            } else {
                content.innerHTML = '<span class="text-red-400">‚ö†Ô∏è Signaal verloren.</span>';
            }

        } catch (error) {
            console.error(error);
            content.innerHTML = '<span class="text-red-400">‚ùå Connectie Fout. API Offline.</span>';
        }
    }

    document.addEventListener('contextmenu', event => event.preventDefault());
    document.getElementById('morphic-query').addEventListener("keypress", function(event) {
        if (event.key === "Enter") sendQuery();
    });
  </script>
