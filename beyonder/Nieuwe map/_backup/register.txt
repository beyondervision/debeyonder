  <script>
    async function processApplication() {
        const btn = document.getElementById('submit-btn');
        const form = document.getElementById('intake-form');
        const msg = document.getElementById('success-msg');
        const codeDisplay = document.getElementById('generated-code');
        const statusFeedback = document.getElementById('status-feedback'); // Gecorrigeerde non-blocking feedback

        const veldnaam = document.getElementById('veldnaam').value;
        const email = document.getElementById('email').value;
        const type = document.getElementById('type').value;

        // 1. Start Animatie
        btn.innerHTML = "TRANSMISSIE STARTEN...";
        btn.classList.add('opacity-70', 'cursor-wait');
        statusFeedback.classList.add('hidden'); // Verberg eerdere fouten

        try {
            // 2. Roep de Generator API aan
            const response = await fetch('/api/veldid/generate', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ veldnaam, email, type })
            });
            
            if (response.status === 404) {
                throw new Error("API niet gevonden. Check of 'api/veldid/generate' actief is.");
            }

            const data = await response.json();

            if (data.success) {
                // SUCCES!
                sessionStorage.setItem('new_cfs_id', data.identityCode);
                sessionStorage.setItem('new_cfs_veld', veldnaam);

                setTimeout(() => {
                    form.classList.add('hidden');
                    codeDisplay.innerText = data.identityCode;
                    msg.classList.remove('hidden');
                }, 1000);
            } else {
                throw new Error("API Fout: " + (data.error || data.message));
            }

        } catch (error) {
            // FOUT AFHANDELING (Non-blocking)
            console.error(error);
            
            btn.innerHTML = "FOUT IN TRANSMISSIE";
            btn.classList.replace('bg-sky-600', 'bg-red-600');
            
            // Toon de non-blocking statusbanner
            statusFeedback.classList.remove('hidden');
            statusFeedback.innerHTML = "‚ùå Transmissie mislukt: " + (error.message || "Onbekende fout.");
            
            setTimeout(() => {
                btn.innerHTML = "START TRANSMISSIE";
                btn.classList.replace('bg-red-600', 'bg-sky-600');
                btn.classList.remove('opacity-70', 'cursor-wait');
                statusFeedback.classList.add('hidden');
            }, 3000);
        }
    }
    document.addEventListener('contextmenu', event => event.preventDefault());
  </script>
