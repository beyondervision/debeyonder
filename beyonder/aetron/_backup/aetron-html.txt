  <script>
    const dash = document.getElementById('dash');
    const overlay = document.getElementById('overlay');
    const enterBtn = document.getElementById('enterBtn');

    const grid = document.getElementById('layersGrid');
    const layers = [...grid.querySelectorAll('[data-layer]')];

    const clusterBtns = [...document.querySelectorAll('.clusterBtn')];
    const focusBtns = [...document.querySelectorAll('.focusBtn')];
    const resetFocus = document.getElementById('resetFocus');

    function setCluster(n){
      layers.forEach((el, idx) => {
        el.style.display = (idx < n) ? '' : 'none';
      });
      // default: no focus, all visible
      clearFocus();
      // UI state
      clusterBtns.forEach(b => b.classList.remove('border-violet-400'));
      const active = clusterBtns.find(b => Number(b.dataset.n) === Number(n));
      if (active) active.classList.add('border-violet-400');
    }

    function clearFocus(){
      layers.forEach(el => {
        el.classList.remove('shadow-dim', 'focus-up');
      });
    }

    function focusLayer(key){
      layers.forEach(el => {
        const isTarget = el.dataset.layer === key;
        if (el.style.display === 'none') return; // ignore hidden
        el.classList.toggle('shadow-dim', !isTarget);
        el.classList.toggle('focus-up', isTarget);
      });
    }

    // Cluster controls
    clusterBtns.forEach(btn => btn.addEventListener('click', () => setCluster(btn.dataset.n)));
    resetFocus.addEventListener('click', clearFocus);

    // Focus buttons in cards
    focusBtns.forEach(btn => btn.addEventListener('click', () => focusLayer(btn.dataset.focus)));

    // Overlay flow (optional)
    enterBtn.addEventListener('click', () => {
      dash.classList.add('fade-out');
      setTimeout(() => {
        dash.style.display = 'none';
        overlay.classList.add('show');
        setTimeout(() => overlay.classList.add('visible'), 30);
      }, 450);
    });

    document.getElementById('overlayClose').addEventListener('click', () => {
      overlay.classList.remove('visible');
      setTimeout(() => {
        overlay.classList.remove('show');
        dash.style.display = '';
        setTimeout(() => dash.classList.remove('fade-out'), 30);
      }, 550);
    });

    document.getElementById('focusLux').addEventListener('click', () => { setCluster(3); focusLayer('luxen'); });
    document.getElementById('focusAet').addEventListener('click', () => { setCluster(3); focusLayer('aetron'); });
    document.getElementById('focusZ3').addEventListener('click', () => { setCluster(3); focusLayer('z3ro'); });

    // Default: 5 layers zichtbaar (triade + contact + Z.A.L)
    setCluster(5);

    // Anti contextmenu (zoals je bestaande stijl)
    document.addEventListener('contextmenu', e => e.preventDefault());
  </script>
